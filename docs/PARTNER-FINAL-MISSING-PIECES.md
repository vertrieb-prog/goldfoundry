### Neue Datenbank:
```sql
CREATE TABLE IF NOT EXISTS partner_report_prefs (
  user_id UUID PRIMARY KEY REFERENCES profiles(id),
  daily_enabled BOOLEAN DEFAULT true,
  daily_time TEXT DEFAULT '08:00',
  weekly_enabled BOOLEAN DEFAULT true,
  monthly_enabled BOOLEAN DEFAULT true,
  notify_new_signup TEXT DEFAULT 'both',
  notify_payment TEXT DEFAULT 'both',
  notify_page_visit TEXT DEFAULT 'push',
  notify_inactivity TEXT DEFAULT 'weekly',
  notify_rank TEXT DEFAULT 'both',
  notify_payout TEXT DEFAULT 'email'
);

CREATE TABLE IF NOT EXISTS hot_leads (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  partner_id UUID REFERENCES profiles(id),
  visitor_hash TEXT,
  source TEXT,
  visit_count INTEGER DEFAULT 1,
  pages_viewed TEXT[],
  signed_up BOOLEAN DEFAULT false,
  signed_up_user_id UUID,
  paid BOOLEAN DEFAULT false,
  first_visit TIMESTAMPTZ DEFAULT now(),
  last_visit TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE IF NOT EXISTS daily_tasks (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES profiles(id),
  date DATE DEFAULT CURRENT_DATE,
  tasks JSONB,
  completed INTEGER DEFAULT 0,
  total INTEGER DEFAULT 5,
  streak_bonus_paid BOOLEAN DEFAULT false
);

CREATE TABLE IF NOT EXISTS training_progress (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES profiles(id),
  level INTEGER DEFAULT 1,
  videos_watched TEXT[],
  quizzes_passed TEXT[],
  certificates TEXT[],
  created_at TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE IF NOT EXISTS community_posts (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES profiles(id),
  content TEXT,
  type TEXT DEFAULT 'post',
  likes INTEGER DEFAULT 0,
  pinned BOOLEAN DEFAULT false,
  created_at TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE IF NOT EXISTS ab_tests (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  partner_id UUID REFERENCES profiles(id),
  element TEXT,
  variant_a TEXT,
  variant_b TEXT,
  visits_a INTEGER DEFAULT 0,
  visits_b INTEGER DEFAULT 0,
  conversions_a INTEGER DEFAULT 0,
  conversions_b INTEGER DEFAULT 0,
  winner TEXT,
  status TEXT DEFAULT 'running',
  created_at TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE IF NOT EXISTS weekly_challenges (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  title TEXT,
  description TEXT,
  reward_fp INTEGER,
  week_start DATE,
  week_end DATE,
  created_at TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE IF NOT EXISTS weekly_challenge_completions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  challenge_id UUID REFERENCES weekly_challenges(id),
  user_id UUID REFERENCES profiles(id),
  completed_at TIMESTAMPTZ DEFAULT now(),
  UNIQUE(challenge_id, user_id)
);

CREATE INDEX idx_hot_leads_partner ON hot_leads(partner_id, last_visit DESC);
CREATE INDEX idx_daily_tasks_user ON daily_tasks(user_id, date);
CREATE INDEX idx_community_posts ON community_posts(created_at DESC);
```
