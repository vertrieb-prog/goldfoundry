## 🔧 TECHNISCHE UMSETZUNG

### 1. Closure Table für unbegrenzte Tiefe:
```sql
-- Bei jedem neuen Referral: Alle Ancestors eintragen
-- User C wird von B geworben, B wurde von A geworben:
INSERT INTO network_tree VALUES
  (A, C, 2),  -- A ist Großeltern (Level 2)
  (B, C, 1),  -- B ist Eltern (Level 1)
  (C, C, 0);  -- C ist sich selbst (Level 0)

-- Query: "Alle User in meinem Netzwerk mit Level"
SELECT descendant_id, depth 
FROM network_tree 
WHERE ancestor_id = 'meine-user-id'
ORDER BY depth;
-- → Sofort alle Level, egal wie tief. Eine Query. <50ms.
```

### 2. Commission Calculation (monatlich):
```typescript
async function calculateCommissions(partnerId: string) {
  const partner = await getPartner(partnerId);
  const tier = getTier(partner.directReferralCount);
  const rates = TIER_RATES[tier.name]; // { 1: 0.20, 2: 0.08, ... 21: 0.0005 }
  
  // Alle User im Netzwerk mit Level
  const network = await supabaseAdmin
    .from("network_tree")
    .select("descendant_id, depth")
    .eq("ancestor_id", partnerId)
    .gt("depth", 0);
  
  let totalFP = 0;
  for (const member of network.data) {
    const rate = rates[member.depth] || rates["unlimited"] || 0;
    if (rate === 0) continue;
    
    // Subscription Revenue
    const sub = await getActiveSubscription(member.descendant_id);
    if (sub) {
      const subFP = Math.round(sub.price * rate * 10); // €→FP
      totalFP += subFP;
      await creditFP(partnerId, subFP, "commission", 
        `L${member.depth} Sub: ${sub.userName} ${sub.planName}`,
        member.descendant_id, member.depth);
    }
    
    // Performance Revenue
    const profit = await getMonthlyProfit(member.descendant_id);
    if (profit > 0) {
      const perfShare = profit * 0.40 * 0.50; // 40% platform × 50% to network
      const perfFP = Math.round(perfShare * rate * 10);
      totalFP += perfFP;
      await creditFP(partnerId, perfFP, "performance",
        `L${member.depth} Perf: ${sub.userName} €${profit}`,
        member.descendant_id, member.depth);
    }
  }
  
  return totalFP;
}
```
