## 🎯 ZIEL: 1.000.000 SEITEN

### Wie kommt man auf 1 Million?

Die Mathematik ist simpel — es ist die KOMBINATION die explodiert:

```
FORMEL:
[Assets] × [Seitentypen] × [Zeiträume] × [Sprachen] = SEITEN

Beispiel:
200 Assets × 8 Seitentypen × 365 Tage × 40 Sprachen = 23.360.000

Wir brauchen nicht mal alle Kombinationen.
Selektiv: 200 × 5 × 52 Wochen × 15 Sprachen = 780.000
Plus Glossar, Tools, Vergleiche, Blog = locker 1.000.000
```

-- CRM Leads & Contacts
CREATE TABLE crm_leads (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id UUID REFERENCES profiles(id),
  email TEXT,
  name TEXT,
  status TEXT DEFAULT 'visitor', -- visitor, registered, trial, active, churned
  score INTEGER DEFAULT 0,       -- Lead Score 0-100
  source TEXT,                   -- organic, referral, partner, telegram, ad
  source_detail TEXT,            -- Welche Seite, welcher Partner
  tags TEXT[],                   -- ["gold", "prop-firm", "beginner"]
  language TEXT DEFAULT 'de',
  country TEXT,
  first_seen TIMESTAMPTZ DEFAULT now(),
  last_seen TIMESTAMPTZ DEFAULT now(),
  last_action TEXT,
  churn_risk INTEGER DEFAULT 0,  -- 0-100
  notes TEXT,
  created_at TIMESTAMPTZ DEFAULT now()
);

-- CRM Activities / Events
CREATE TABLE crm_activities (
  id BIGSERIAL PRIMARY KEY,
  user_id UUID,
  lead_id UUID REFERENCES crm_leads(id),
  type TEXT NOT NULL,            -- page_view, login, trade, email_open, support_ticket
  detail JSONB,                  -- { page: "/kurs/bitcoin", duration: 45 }
  created_at TIMESTAMPTZ DEFAULT now()
);

-- CRM Email Campaigns
CREATE TABLE crm_campaigns (
  id BIGSERIAL PRIMARY KEY,
  name TEXT NOT NULL,
  type TEXT,                     -- welcome, winback, upgrade, retention, seasonal
  segment JSONB,                 -- { status: "registered", days_inactive: 7 }
  subject TEXT,
  template TEXT,
  sent_count INTEGER DEFAULT 0,
  open_count INTEGER DEFAULT 0,
  click_count INTEGER DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT now()
);

-- CRM Email Queue
CREATE TABLE crm_email_queue (
  id BIGSERIAL PRIMARY KEY,
  user_id UUID,
  campaign_id BIGINT REFERENCES crm_campaigns(id),
  email TEXT NOT NULL,
  subject TEXT NOT NULL,
  body TEXT NOT NULL,
  status TEXT DEFAULT 'pending', -- pending, sent, opened, clicked, bounced
  scheduled_at TIMESTAMPTZ,
  sent_at TIMESTAMPTZ,
  opened_at TIMESTAMPTZ
);

-- Support Tickets
CREATE TABLE support_tickets (
  id BIGSERIAL PRIMARY KEY,
  user_id UUID,
  category TEXT,                 -- broker, copier, billing, technical, general
  priority TEXT DEFAULT 'medium', -- low, medium, high, critical
  subject TEXT NOT NULL,
  description TEXT,
  status TEXT DEFAULT 'open',    -- open, in_progress, resolved, closed
  assigned_to TEXT,
  resolution TEXT,
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now(),
  resolved_at TIMESTAMPTZ
);

-- Affiliate Payouts
CREATE TABLE affiliate_payouts (
  id BIGSERIAL PRIMARY KEY,
  partner_id UUID REFERENCES profiles(id),
  amount NUMERIC NOT NULL,
  currency TEXT DEFAULT 'EUR',
  period_start DATE,

  payment_method TEXT,           -- crypto_btc, crypto_usdt, bank_transfer
  payment_ref TEXT,
  paid_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT now()
);

-- Indexes
